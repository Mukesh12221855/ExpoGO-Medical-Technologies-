import AOS from 'aos';

import mriScanner from '../assets/products/mriScanner.jpg';
import patientMonitor from '../assets/products/patientMonitor.jpg';
import portableVentilator from '../assets/products/portableVentilator.jpg';

function Products() {
  useEffect(() => {
    AOS.init({
  duration: 1000,
  once: false,        // Allow animation on every scroll
  mirror: true,       // Animate when scrolling up
  easing: 'ease-in-out',  // Smooth easing
});

    // ‚úÖ Search & Filter States
  const [searchQuery, setSearchQuery] = useState('');
  const [categoryFilter, setCategoryFilter] = useState('');

  const [isCartOpen, setIsCartOpen] = useState(false);

// ‚úÖ Cart State
const [cart, setCart] = useState([]);
// ‚úÖ Add to Cart Function
const addToCart = (product) => {
  setCart((prevCart) => {
    const existingProduct = prevCart.find((item) => item.name === product.name);
    if (existingProduct) {
      // Increase quantity if product already exists
      return prevCart.map((item) =>
        item.name === product.name ? { ...item, quantity: item.quantity + 1 } : item
      );
    } else {
      // Add new product with quantity 1
      return [...prevCart, { ...product, quantity: 1 }];
    }
  });
};
const increaseQuantity = (productName) => {
  setCart((prevCart) =>
    prevCart.map((item) =>
      item.name === productName ? { ...item, quantity: item.quantity + 1 } : item
    )
  );
};

const decreaseQuantity = (productName) => {
  setCart((prevCart) =>
    prevCart
      .map((item) =>
        item.name === productName
          ? { ...item, quantity: item.quantity - 1 }
          : item
      )
      .filter((item) => item.quantity > 0) // remove if quantity becomes 0
  );
};


//Remove from cart
const removeFromCart = (indexToRemove) => {
  setCart((prevCart) => prevCart.filter((_, index) => index !== indexToRemove));
};

];

 // ‚úÖ Filtered Products

    const matchesSearch = product.name.toLowerCase().includes(searchQuery.toLowerCase()) ;

    const matchesCategory = categoryFilter === '' || product.category === categoryFilter;

    return matchesSearch && matchesCategory;
  });
  
  return (
    <section id="products" style={styles.productsSection}>

<div style={styles.cartBar} data-aos="fade-up">

  <button style={styles.checkoutIcon}>üí≥ Checkout</button>
</div>
      <p data-aos="fade-up" style={styles.description}>
        Explore our wide range of critical-care medical equipment, designed for affordability and quality.  
        Whether you need ventilators, patient monitors, or infusion pumps, we provide sustainable solutions for every healthcare setting.
      </p>
 {/* Search and Filter */}

        <input

          onChange={(e) => setSearchQuery(e.target.value)}
          style={styles.searchInput}

          onChange={(e) => setCategoryFilter(e.target.value)}

        </select>
      </div>

{/* Products List */}
<div style={styles.productsWrapper} data-aos="fade-up">
  {filteredProducts.map((product, index) => (

        src={product.image}
        alt={product.name}
        style={styles.productImage}
      />
      <h3 style={styles.productName}>{product.name}</h3>
      <p style={styles.productDesc}>{product.description}</p>
      <p style={styles.productCategory}>Category: {product.category}</p>
      <p style={styles.productPrice}>{product.price}</p>
       {/* New Buttons */}
  <div style={styles.buttonGroup}>
    <button style={styles.addToCartButton}onClick={() => addToCart(product)}> üõí Add to Cart</button>
    <button style={styles.buyNowButton}>üõçÔ∏è Buy Now</button>
  </div>
    </div>
    
  ))}
</div>
{/* ‚úÖ Add This Cart Sidebar just before </section> */}
<div style={{ ...styles.cartSidebar, right: isCartOpen ? '0' : '-400px' }}>
  <h3 style={styles.cartTitle}>üõí Your Cart</h3>

  {cart.length === 0 ? (
    <p style={styles.emptyCartText}>Your cart is empty.</p>
  ) : (
    <>
      <ul style={styles.cartItemList}>
        {cart.map((item, index) => (
          <li key={index} style={styles.cartItem}>
            <img src={item.image} alt={item.name} style={styles.cartItemImage} />
            <div>
              <p style={styles.cartItemName}>{item.name}</p>
              <p style={styles.cartItemPrice}>{item.price}</p>

              {/* Quantity Controls */}
              <div style={styles.quantityControls}>
                <button onClick={() => decreaseQuantity(item.name)} style={styles.quantityButton}>‚ûñ</button>
                <span style={styles.quantityText}>{item.quantity}</span>
                <button onClick={() => increaseQuantity(item.name)} style={styles.quantityButton}>‚ûï</button>
              </div>
            </div>

            <button onClick={() => removeFromCart(index)} style={styles.removeButton}>‚ùå</button>
          </li>
        ))}
      </ul>

      {/* ‚úÖ Checkout Button */}
      <button
        style={styles.checkoutButton}
        onClick={() => {
          setIsCartOpen(false); // Close the cart
          navigate('/checkout'); // Navigate to checkout
        }}
      >
        ‚úÖ Proceed to Checkout
      </button>
    </>
  )}
</div>


    </section>
    
  );
}
